    StartDelay: byte;     { тайм-аут пуска 0 - 60 сек (1 байт) }

    PumpCount: byte;      { количество насосов, необходимых для тушения 1 - 8 шт.  }
    LineCount: byte;      { число записей в переменной части }
  end;
